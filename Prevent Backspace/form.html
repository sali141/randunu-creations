Prevent Backspace 

<script>
$(document).keydown(function (e) {
                var element = e.target.nodeName.toLowerCase();
                if ((element != 'input' && element != 'textarea')) {
                    if (e.keyCode === 8) {
                        e.preventDefault();
                        return false;
                    }
                } </script>
